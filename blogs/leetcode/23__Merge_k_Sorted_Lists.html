leetcode 023: Merge k Sorted Lists



https://leetcode.com/problems/generate-parentheses/

problem: 
Merge k sorted linked lists and return it as one sorted list. Analyze and describe its complexity.

Example:

Input:
[
  1->4->5,
  1->3->4,
  2->6
]
Output: 1->1->2->3->4->4->5->6

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def mergeKLists(self, lists: List[ListNode]) -> ListNode:
        listSize = len(lists)
        for i in range(listSize -1, -1, -1):
            if lists[i] is None:
                del lists[i]
                
        head, tail = None, None
        while len(lists) > 0:
            minVal = lists[0].val
            minIdx = 0
            for i, h in enumerate(lists):
                if h.val < minVal:
                    minVal = h.val
                    minIdx = i
            if head is None:
                head = lists[minIdx]
                tail = head
            else:
                tail.next = lists[minIdx]
                tail = lists[minIdx]
            lists[minIdx] = lists[minIdx].next
            if lists[minIdx] is None:
                del lists[minIdx]
        return head
