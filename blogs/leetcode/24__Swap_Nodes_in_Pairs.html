You may not modify the values in the list's nodes, only nodes itself may be changed.

 

Example:

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        if (head == NULL)
            return head;
        
        ListNode *tail2 = NULL;
        ListNode *head2 = NULL;
        while (head && head->next){
            ListNode *n1 = head;
            ListNode *n2 = head->next;
            head = head->next->next;
            if (head2 == NULL){
                head2 = n2;
                head2->next = n1;
                tail2 = head2->next;
            }else{
                tail2->next = n2;
                n2->next = n1;
                tail2 = n1;
            }
        }
        if (head2 == NULL)
            head2 = head;
        else
            tail2->next = head;
        return head2;
        
    }
};
