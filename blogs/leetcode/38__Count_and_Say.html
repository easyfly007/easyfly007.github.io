leetcode 038. count and say



https://leetcode.com/problems/count-and-say/

problem: 
The count-and-say sequence is the sequence of integers with the first five terms as following:

1.     1
2.     11
3.     21
4.     1211
5.     111221
1 is read off as "one 1" or 11.
11 is read off as "two 1s" or 21.
21 is read off as "one 2, then one 1" or 1211.

Given an integer n where 1 ≤ n ≤ 30, generate the nth term of the count-and-say sequence.

Note: Each term of the sequence of integers will be represented as a string.

 

Example 1:

Input: 1
Output: "1"
Example 2:

Input: 4
Output: "1211"
start code (c++) 
class Solution {
public:
    string countAndSay(int n) {
    }
}
solution:

我的思路：


class Solution {
public:
    string countAndSay(int n) {
        if (n == 1)
            return "1";
        
        string str = "1";
        for (int i = 2; i <= n; i ++){
            str = countAndSayStr(str);
            cout << " i = " << i << ", str = " << str << endl;
        }
        return str;
    }

    string sayIt(char what, int count){
        std::ostringstream oss;
        oss << count;
        oss << what;
        return oss.str();
    }

    string countAndSayStr(string &str){
        if (str.size() == 0)
            return "";
        if (str.size() == 1)
            return "1" + str;
        
        string solution;
        int i = 1;
        int count = 1;
        while (i < str.size()){
            // i is the begining of a new char
            if (str[i] != str[i -1]){
                string temp = sayIt(str[i-1], count);
                solution += temp;
                count = 1;
            }else{
                count ++;
            }
            i++;
        }
        string temp2 = sayIt(str[str.size()-1], count);
        solution += temp2;
        return solution;
    }
            
};
