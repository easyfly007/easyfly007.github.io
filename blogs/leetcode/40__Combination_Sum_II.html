039. Combination Sum



https://leetcode.com/problems/combination-sum-ii/

problem: 
Given a collection of candidate numbers (candidates) and a target number (target), find all unique combinations in candidates where the candidate numbers sums to target.

Each number in candidates may only be used once in the combination.

Note:

All numbers (including target) will be positive integers.
The solution set must not contain duplicate combinations.
Example 1:

Input: candidates = [10,1,2,7,6,1,5], target = 8,
A solution set is:
[
  [1, 7],
  [1, 2, 5],
  [2, 6],
  [1, 1, 6]
]
Example 2:

Input: candidates = [2,5,2,1,2], target = 5,
A solution set is:
[
  [1,2,2],
  [5]
]

class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        candidates = sorted(candidates, reverse = True)
        find, solution = self.combinationSum2Call(candidates, target)
        if find:
            # need to remove duplicate 
            size = len(solution)
            for i in range(size-1, 0, -1):
                if solution[i] in solution[0:i]:
                    del solution[i]
            return solution
        return []
    
    def combinationSum2Call(self, candidates, target):
        if target < 0:
            return False, None
        if target == 0:
            return True, [[],]
        if len(candidates) == 0:
            return False, None
        
        solution = []
        for i in range(len(candidates)):
            val = candidates[i]
            find, temp_solution = self.combinationSum2Call(candidates[i+1: ], target - val)
            if find:
                for x in temp_solution:
                    x.append(val)
                    solution.append(x)
        return len(solution) >0, solution
