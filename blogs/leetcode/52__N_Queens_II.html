leetcode 52: n queen II



https://leetcode.com/problems/n-queens-ii/

problem: 
The n-queens puzzle is the problem of placing n queens on an n×n chessboard such that no two queens attack each other.



Given an integer n, return the number of distinct solutions to the n-queens puzzle.

Example:

Input: 4
Output: 2
Explanation: There are two distinct solutions to the 4-queens puzzle as shown below.
[
 [".Q..",  // Solution 1
  "...Q",
  "Q...",
  "..Q."],

 ["..Q.",  // Solution 2
  "Q...",
  "...Q",
  ".Q.."]
]

solution:

利用深度优先遍历 

class Solution(object):
    def totalNQueens(self, n):
        """
        :type n: int
        :rtype: int
        """
        board = [['.'] * n for i in range(n)]
        self.solution = 0
        self.n = n
        self.queen_cnt = 0
        self.filling_from_pos(board, 0, 0)
        return self.solution
    
    def filling_from_pos(self, board, i, j):
        if i > self.n -1 or j > self.n -1:
            if self.queen_cnt == self.n:
                self.solution += 1
            return
        for x in ['Q', '.']:
            board[i][j] = x
            if x == 'Q':
                self.queen_cnt += 1
            if self.is_valid(board, i, j):
                i2, j2 = self.get_next_pos(board, i, j)
                self.filling_from_pos(board, i2, j2)
            board[i][j] = '.'
            if x == 'Q':
                self.queen_cnt -= 1
    
    def is_valid(self, board, i, j):
        queen_cnt = 0
        for x in range(i+1):
            if board[x][j] == 'Q':
                queen_cnt += 1
        if queen_cnt > 1:
            return False
        if queen_cnt == 0 and i == self.n -1:
            return False
        
        queen_cnt = 0
        for x in range(j+1):
            if board[i][x] == 'Q':
                queen_cnt += 1
        if queen_cnt > 1:
            return False
        if queen_cnt == 0 and j == self.n -1:
            return False
        
        i0, j0 = i, j
        queen_cnt = 0
        while i >=0 and j >= 0:
            if board[i][j] == 'Q':
                queen_cnt += 1
            i, j = i-1, j-1
        if queen_cnt > 1:
            return False
        
        i, j = i0, j0
        queen_cnt = 0
        while i >= 0 and j<= self.n -1:
            if board[i][j] == 'Q':
                queen_cnt += 1
            i, j = i -1, j + 1
        if queen_cnt > 1:
            return False
        
        return True
            
    
    def get_next_pos(self, board, i, j):
        if j == self.n -1:
            return i+1, 0
        return i, j +1
