leetcode 65. Valid Number



https://leetcode.com/problems/valid-number/

problem: 
Validate if a given string can be interpreted as a decimal number.

Some examples:
"0" => true
" 0.1 " => true
"abc" => false
"1 a" => false
"2e10" => true
" -90e3   " => true
" 1e" => false
"e3" => false
" 6e-1" => true
" 99e2.5 " => false
"53.5e93" => true
" --6 " => false
"-+3" => false
"95a54e53" => false

Note: It is intended for the problem statement to be ambiguous. You should gather all requirements up front before implementing one. However, here is a list of characters that can be in a valid decimal number:

Numbers 0-9
Exponent - "e"
Positive/negative sign - "+"/"-"
Decimal point - "."
Of course, the context of these characters also matters in the input.

class Solution(object):
    def isNumber(self, s):
        """
        :type s: str
        :rtype: bool
        """
        beg = 0
        for i in range(len(s)):
            if s[i] != ' ':
                beg = i
                break
        end = len(s) -1
        for i in range(len(s) -1, -1, -1):
            if s[i] != ' ':
                end = i
                break
        s = s[beg: end +1]
                
        pos_e = s.find('e')
        if pos_e == -1:
            return self.isNumberFirst(s)
        else:
            return self.isNumberFirst(s[:pos_e]) and self.isNumberSecond(s[pos_e+1:])
    
    def isNumberFirst(self, s):
        print('is number first : >' + s + '<')
        if len(s) == 0:
            return False
        # only accept ([+-] xx.xx)
        if s[0] in '-+':
            s = s[1:]
        if len(s) == 0:
            return False
        pos_dot = s.find('.')
        if pos_dot != -1:
            s1 = s[:pos_dot]
            s2 = s[pos_dot +1 :]
            if len(s1) == 0 and len(s2) == 0:
                return False
            for i in s1:
                if i not in '0123456789':
                    return False
            for i in s2:
                if i not in '0123456789':
                    return False
        else:
            for i in s:
                if i not in '0123456789':
                    return False
        return True
    
    def isNumberSecond(self, s):
        print('is number second: >' + s + '<')
        if len(s) == 0:
            return False
        if s[0] in '-+':
            s = s[1:]
        if len(s) == 0:
            return False
        for x in s:
            if x not in '0123456789':
                return False
        return True
